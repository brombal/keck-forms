import { ObjectPath } from "/src/util/ObjectPath.types";
import { get as _get } from "lodash-es";
 
function isEmptyPath(path: any): boolean {
  return (
    path === undefined ||
    path === null ||
    path === "" ||
    (Array.isArray(path) && isEmptyPath(path[0]))
  );
}
 
export default function get<T extends object>(obj: T, path: ObjectPath<T> | null | undefined) {
  return isEmptyPath(path) ? obj : _get(obj, path as any);
}
